# PyCraft — Hello World
 
from reflex import rx
 
class State(rx.State):
  name: str = "World"
 
def index():
  return rx.vstack(
    rx.heading(f"Hello, {State.name}!"),
    rx.input(
      placeholder="Your name...",
      on_change=State.set_name
    )
  )
 
app = rx.App()
app.add_page(index, route="/")
